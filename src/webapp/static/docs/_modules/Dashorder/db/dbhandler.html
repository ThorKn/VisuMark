    def __init__(self, username: str = None, password: str = None,
                 ip: str = None, port: int = None, dbname: str = 'maldaner',
                 article_template: str = None,
                 customer_template: str = None,
                 order_template: str = None) -> None:
        # Configure logging module name
        # logging.setLogger(self.__class__.__name__)

        # Construct connection uri

        connection_data = {}

        if ip is not None:
            connection_data['host'] = ip
            connection_data['port'] = 27017

        if port is not None:
            connection_data['port'] = port

        if username is not None:
            connection_data['username'] = username
            connection_data['password'] = ''
            connection_data['authSource'] = dbname

        if password is not None:
            connection_data['password'] = password

        # If connect uri is empty, use autodiscover
        if connection_data:
            ip = connection_data['host'] if 'host' in connection_data else 'localhost'
            port = connection_data['port'] if 'port' in connection_data else 27017

            if 'username' in connection_data:
                logging.debug('Using MongoDB user {} for database {}'
                              .format(connection_data['username'], connection_data['authSource']))

            mongo = MongoClient(**connection_data)
            logging.debug('MongoDB connected to {}:{}'.format(ip, port))
        else:
            mongo = MongoClient()

        self._db = mongo[dbname]
        logging.info('Connected to database "{}" @ "{}"'.format(dbname, str(mongo.address)))

        # Load JSON data templates (specifies relevant keys from BillBee requests)
        base_path = Path(__file__).parent

        if (customer_template is None):
            customer_template = (base_path / '../config/customer.json').resolve()

        with open(customer_template, 'r') as customer:
            self.customer_template = jload(customer)

        if (article_template is None):
            article_template = (base_path / '../config/article.json').resolve()

        with open(article_template, 'r') as article:
            self.article_template = jload(article)

        if (order_template is None):
            order_template = (base_path / '../config/order.json').resolve()

        with open(order_template, 'r') as order:
            self.order_template = jload(order)

        # Variables that will contain a unique key name for each collection
        self._keys = {}

        try:
            # Initialize Collections (if not done yet)
            self._initCollections()
        except OperationFailure as e:
            if e.code == PYMONGO_AUTHENTICATION_ERROR:
                # Authentication failed? Retry without authentication
                logging.info('Authentication for user "{}" failed, retrying without authentication'.format(username))
                mongo.close()
                del connection_data['username']
                del connection_data['password']
                del connection_data['authSource']
                mongo = MongoClient(**connection_data)
                self._db = mongo[dbname]
                self._initCollections()

        logging.info('Database successfully initialized')

    def __initCollection(self, name: str, config: dict) -> None:
        # If a Collation is specified, use it
        if 'Collation' in config:
            collation = Collation(**config['Collation'])
            self._db.create_collection(name, collation=collation)
            logging.debug('Created Collation for "{}" collection: {}'.format(name, str(collation)))
        else:
            self._db.create_collection(name)

        # Create indices
        if 'Indices' in config:
            indices = []

            for index in config['Indices']:
                if 'Options' in index:
                    collation = None

                    if 'Collation' in index['Options']:
                        collation = Collation(**index['Options']['Collation'])
                        del index['Options']['Collation']
                        model = IndexModel(index['Name'], **index['Options'],
                                           collation=collation)
                    else:
                        model = IndexModel(index['Name'], **index['Options'])

                    # Store the name of one unique key
                    if 'unique' in index['Options'] and name not in self._keys:
                        self._keys[name] = index['Name']
                else:
                    model = IndexModel(index['Name'])

                indices.append(model)
                logging.debug('Created IndexModel for "{}" collection: {}'.format(
                              name, str(model.document)))

            self._db[name].create_indexes(indices)
            logging.debug('Indices for "{}" collection updated'.format(name))

        logging.info(name + ' collection created and configured')

    def __addKey(self, collection_name: str) -> None:
        indices = self._db[collection_name].index_information()

        if len(indices) == 1:
            self._keys[collection_name] = indices[indices.keys()[0]]['key'][0][0]
        else:
            for key in list(indices.keys())[1:]:
                if 'unique' in indices[key] and indices[key]['unique']:
                    self._keys[collection_name] = indices[key]['key'][0][0]
                    return

    def _initCollections(self) -> None:
        # Create collections if not existant yet
        collection_names = self._db.list_collection_names()

        if 'customers' not in collection_names:
            self.__initCollection('customers', self.customer_template['DashOrderConfiguration'])
        else:
            self.__addKey('customers')

        if 'orders' not in collection_names:
            self.__initCollection('orders', self.order_template['DashOrderConfiguration'])
        else:
            self.__addKey('orders')

        if 'articles' not in collection_names:
            self.__initCollection('articles', self.article_template['DashOrderConfiguration'])
        else:
            self.__addKey('articles')

        if 'Seller' in self.order_template:
            if 'sellers' not in collection_names:
                self._db.create_collection('sellers')
                self._db.sellers.create_index('BillbeeShopId', unique=True)

            self._keys['sellers'] = 'BillbeeShopId'

        if 'Category1' in self.article_template or 'Category2' in self.article_template\
           or 'Category3' in self.article_template:
            if 'categories' not in collection_names:
                self._db.create_collection('categories')
                self._db.categories.create_index('Id', unique=True)

            self._keys['categories'] = 'Id'

        if 'Tags' in self.article_template:
            if 'tags' not in collection_names:
                self._db.create_collection('tags')
                self._db.tags.create_index('Name', unique=True)

            self._keys['tags'] = '_id'

        # Remove DashOrder config from data models
        del self.customer_template['DashOrderConfiguration']
        del self.order_template['DashOrderConfiguration']
        del self.article_template['DashOrderConfiguration']

    def __filterData(self, data: dict, template: dict, result: dict = None) -> dict:
        if result is None:
            result = {}

        if isinstance(data, list):
            for idx, elem in enumerate(data):
                result.append({})
                self.__filterData(elem, template[0], result[idx])
        else:
            for entry_key in data:
                if entry_key in template:
                    if isinstance(data[entry_key], dict):
                        result[entry_key] = {}
                        self.__filterData(data[entry_key], template[entry_key], result[entry_key])
                    elif isinstance(data[entry_key], list):
                        result[entry_key] = []
                        self.__filterData(data[entry_key], template[entry_key], result[entry_key])
                    else:
                        result[entry_key] = data[entry_key]

        return result

    def __bulkUpsert(self, data: List[dict], collection_name: str, unique_index: Any) -> BulkWriteResult:
        # Prepare list for bulk upsert request
        bulk_data = [
            UpdateOne(
                {unique_index: entry[unique_index]},
                {'$setOnInsert': entry},
                upsert=True
            ) for entry in data
        ]
        # bulk_write the upsert data and log the result
        result = self._db[collection_name].bulk_write(bulk_data)
        logging.info('Did{}update collection "{}"'.format(' ' if result else ' NOT ',
                                                          collection_name))
        return result

    def __ordersFilterHelper(self, articles: List[Union[int, str]] = [], shops: List[Union[int, str]] = None,
                             customers: List[Union[int, str]] = None):
        # apply filters
        conditions = []

        if articles:
            conditions.append({'$in': ['$OrderItems.Product.BillbeeId', articles]})
        if shops:
            conditions.append({'$in': ['$BillbeeShopId', shops]})
        if customers:
            conditions.append({'$in': ['$Customer', customers]})

        return conditions

    def __ordersSortHelper(self, sort):
        sort_order = {}

        for code in sort:
            if code == 'a':
                sort_order['ArticleId'] = ASCENDING
            elif code == 's':
                sort_order['ShopId'] = ASCENDING
            elif code == 'c':
                sort_order['CustomerId'] = ASCENDING
            elif code == 'd':
                sort_order['CreatedAt'] = ASCENDING
            elif code == 't':
                sort_order['TagId'] = ASCENDING
            elif code == 'cat':
                sort_order['CategoryId'] = ASCENDING

        return sort_order

[docs]    def getOrdersMultiFilter(self, articles: List[Union[int, str]] = [], shops: List[Union[int, str]] = None,
                             customers: List[Union[int, str]] = None, sort: List[str] = []):
        '''
        Get orders, limit them by multiple filters for articles, shops and customers.
        sort can contain 'a' for articles, 's' for shops and 'c' for customers and
        'd' for date in any order. The order of the elements in sort defines the sort order
        '''

        pipeline = [
            {'$unwind': '$OrderItems'}
        ]

        # apply filters
        conditions = self.__ordersFilterHelper(articles=articles, shops=shops, customers=customers)

        if conditions:
            pipeline += [
                {'$match': {  # Remove orders containing invalid order ids
                    '$expr': {
                        '$and': conditions
                     }
                   }
                 }
            ]

        pipeline += [
            {'$addFields': {
                    'ArticleId': '$OrderItems.Product.BillbeeId',
                    'CustomerId': '$Customer',
                    'ShopId': '$BillbeeShopId',
                    'Price': '$OrderItems.TotalPrice',
                    'CreatedAt': '$CreatedAt'
                }
             },
            {'$project': {
                    'ArticleId': 1,
                    'CustomerId': 1,
                    'ShopId': 1,
                    'Price': 1,
                    'CreatedAt': 1
                }
             }
        ]

        if sort:
            sort_order = self.__ordersSortHelper(sort)
            pipeline += [{'$sort': sort_order}]

        return self._db.orders.aggregate(pipeline)



[docs]    def getOrdersContainingTags(self, tags, articles: List[Union[int, str]] = [],
                                shops: List[Union[int, str]] = None,
                                customers: List[Union[int, str]] = None, sort: List[str] = ['t', 'd']):
        '''
        Get orders, optionally limiting them by a list of tag ids
        '''

        otags = [ObjectId(tag) for tag in tags]
        pipeline = [
            {'$unwind': '$OrderItems'},  # Create a new order document for each sold product within an order
            {'$lookup': {  # Join with articles
                'from': 'articles',
                'localField': 'OrderItems.Product.BillbeeId',
                'foreignField': 'Id',
                'as': 'articles'
                }
             }
        ]

        # apply filters
        conditions = []

        if otags:
            conditions.append({'$anyElementTrue': [{'$setIntersection': ['$$article.Tags', otags]}]})

        conditions.extend(self.__ordersFilterHelper(articles=articles, shops=shops, customers=customers))

        if conditions:
            pipeline += [{'$project': {  # Remove articles not containing the relevant tags
                 'articles': {
                    '$filter': {
                       'input': '$articles',
                       'as': 'article',
                       'cond': {'$and': conditions}
                    }
                 },
                 'CreatedAt': 1,
                 'OrderItems.TotalPrice': 1,
                 'OrderItems.Product.BillbeeId': 1,
                 'Customer': 1,
                 'BillbeeShopId': 1
              }
             }]

        pipeline += [
            {'$unwind': '$articles'},  # Create document for each article
            {'$addFields': {
                    'CreatedAt': '$CreatedAt',
                    'Price': '$OrderItems.TotalPrice',
                    'TagId': '$articles.Tags',
                    'ArticleId': '$OrderItems.Product.BillbeeId',
                    'CustomerId': '$Customer',
                    'ShopId': '$BillbeeShopId'
                }
             },
            {'$unwind': '$TagId'},  # Create document for each tag
        ]

        if otags:
            pipeline += [
                {'$match': {  # Remove orders containing invalid tags
                    '$expr': {
                        '$in': ['$TagId', otags]
                     }
                   }
                 }
            ]

        pipeline += [
            {'$project': {
                    'ArticleId': 1,
                    'CustomerId': 1,
                    'ShopId': 1,
                    'Price': 1,
                    'CreatedAt': 1,
                    'TagId': 1
                }
             }
        ]

        if sort:
            sort_order = self.__ordersSortHelper(sort)
            pipeline += [{'$sort': sort_order}]

        return self._db.orders.aggregate(pipeline)

[docs]    def getOrdersContainingZips(self, zips, sort=['c']):
        '''
        Get orders, optionally limiting them by a list of zip codes.
        The elements in zips are regarded as the first letters a zip code must contain.
        sort can contain 'a' for articles, 's' for shops and 'z' for zips and
        'd' for date in any order. The order of the elements in sort defines the sort order
        '''

        # Make valid regex expression
        zips = ['^' + str(zip) + '.*|' for zip in zips]
        zips = ''.join(zips)[:-1]

        pipeline = [
            {'$unwind': '$OrderItems'},  # Create a new order document for each sold product within an order
            {'$lookup': {  # Join with customers
                'from': 'customers',
                'localField': 'Customer',
                'foreignField': 'Id',
                'as': 'customers'
                }
             }
        ]

        if zips:
            pipeline += [
                {'$match': {  # Remove orders containing invalid categories
                    'customers.Zip': {'$regex': zips}
                   }
                 }
            ]

        pipeline += [
            {'$unwind': '$customers'},  # Create document for each customer
            {'$addFields': {
                    'ArticleId': '$OrderItems.Product.BillbeeId',
                    'Price': '$OrderItems.TotalPrice',
                    'Zip': '$customers.Zip'
                }
             },
            {'$project': {
                'ArticleId': 1,
                'CreatedAt': 1,
                'Price': 1,
                'Zip': 1
                }
             }
        ]

        if sort:
            sort_order = {}

            for code in sort:
                if code == 'a':
                    sort_order['ArticleId'] = ASCENDING
                elif code == 'z':
                    sort_order['Zip'] = ASCENDING
                elif code == 'd':
                    sort_order['CreatedAt'] = ASCENDING

            pipeline += [{'$sort': sort_order}]

        return self._db.orders.aggregate(pipeline)

        ]

        Get orders, optionally limiting them by a list of shop ids
        '''

        pipeline = [
            {'$unwind': '$OrderItems'}
        ]

        if shops:
            pipeline += [
                {'$redact': {
                    '$cond': {
                            'if': {'$in': ['$BillbeeShopId', shops]},
                            'then': '$$KEEP',
                            'else': '$$PRUNE'
                        }
                    }
                 }
            ]

        pipeline += [
            {'$addFields': {
                    'ShopId': '$BillbeeShopId',
                    'Price': '$OrderItems.TotalPrice',
                }
             },
            {'$project': {
                    'CreatedAt': 1,
                    'ShopId': 1,
                    'Price': 1
                }
             },
            {'$sort': {'ShopId': ASCENDING, 'CreatedAt': ASCENDING}}
        ]

        return self._db.orders.aggregate(pipeline)
