    def _zin(self, ychoice: str, gchoice: str, gvalues: List[Union[int, str]], orderfunc: Callable,
             itemfunc: Union[Callable, None], order_itemkey: str, item_key: str, item_evalname: str):
        # Get orders and relevant items from database
        orders = orderfunc(gvalues)
        item = itemfunc(gvalues) if itemfunc is not None else None
        result = []
        # TODO: next() without ExceptionHandling (empty list, end of list) is dangerous!
        lastitem = next(item) if item is not None else None
        dates = []
        sales = []

        # For each item, accumulate the revenue in total or per day
        for order in orders:
            if gchoice != 'ALL' and lastitem[item_key] != order[order_itemkey]:
                result.append({
                    'x': dates,
                    'y': sales,
                    'type': div_tree['ZIN'][ychoice][gchoice]['type'],
                    'name': 'ALL' if gchoice == 'ALL' else eval('lastitem' + item_evalname)
                })
                dates = []
                sales = []
                lastitem = next(item)

            if not len(dates):
                dates.append(order['CreatedAt'])

                if ychoice in ['BAN', 'BAT']:
                    sales.append(1)
                else:
                    if gchoice == 'ALL':
                        sales.append(order['OrderItems']['TotalPrice'])
                    else:
                        sales.append(order['Price'])
            elif dates[-1] != order['CreatedAt']:
                dates.append(order['CreatedAt'])

                # Accumulate in total
                if ychoice == 'UMS':
                    sales.append(sales[-1] + (order['OrderItems']['TotalPrice']
                                 if gchoice == 'ALL' else order['Price']))
                # Accumulate per day
                elif ychoice == 'UMT':
                    sales.append(order['OrderItems']['TotalPrice'] if gchoice == 'ALL' else order['Price'])
                elif ychoice == 'BAN':
                    sales.append(sales[-1] + 1)
                elif ychoice == 'BAT':
                    sales.append(1)
            else:
                if ychoice in ['BAN', 'BAT']:
                    sales[-1] += 1
                else:
                    sales[-1] += order['OrderItems']['TotalPrice'] if gchoice == 'ALL' else order['Price']

        result.append({
            'x': dates,
            'y': sales,
            'type': div_tree['ZIN'][ychoice][gchoice]['type'],
            'name': 'ALL' if gchoice == 'ALL' else eval('lastitem' + item_evalname)
        })

        return result
