[docs]@bp.route('/einstellungen', methods=['GET', 'POST'])
@login_required
def einstellungen():
    form_bb = BillbeeForm()
    form_up = UpdateForm()
    form_da = DashForm()
    if form_bb.validate_on_submit() and form_bb.bb_submit.data:
        bb_api_user = form_bb.bb_username.data
        bb_api_pw = form_bb.bb_password.data
        bb_api_key = form_bb.bb_apikey.data
        bbhandler = BbHandler(username=bb_api_user,
                              password=bb_api_pw,
                              apikey=bb_api_key)
        test_conn = bbhandler.test_connection()
        if not test_conn:
            flash('Verbindung zu Billbee ist fehlgeschlagen. Die Daten wurden nicht gespeichert')
        else:
            if current_user.is_authenticated:
                current_user.billbee_username = bb_api_user
                current_user.billbee_password = bb_api_pw
                current_user.billbee_apikey = bb_api_key
                db.session.commit()
                flash('Verbindung zu Billbee war erfolgreich und die Daten wurden gespeichert.')
    if form_up.validate_on_submit() and form_up.bb_update.data:
        print("up_save pressed and validated.")
        bbhandler = BbHandler()
        test_conn = bbhandler.test_connection()
        if not test_conn:
            flash('Verbindung zu Billbee ist fehlgeschlagen. Sind die richtigen Daten gespeichert?')
        else:

            dbhandler = DBHandler(username=MONGO_USER,
                                  password=MONGO_PW,
                                  dbname=MONGO_DBNAME)
            # Update MongoDB with the data gathered from Billbee
            customers = bbhandler.get_customers()
            if len(customers) > 0:
                dbhandler.updateCustomers(customers)
            time.sleep(2)
            articles = bbhandler.get_articles()
            if len(articles) > 0:
                dbhandler.updateArticles(articles)
            time.sleep(2)
            orders = bbhandler.get_orders()
            if len(orders) > 0:
                dbhandler.updateOrders(orders)
            flash('Update erfolgreich')
    if form_da.validate_on_submit() and form_da.da_submit.data:
        cfg_graph.font_size_axis = form_da.dash_fontsize_axis.data
        cfg_graph.font_size_legend = form_da.dash_fontsize_axis.data
        cfg_graph.margin_left = form_da.dash_margin_left.data
        cfg_graph.margin_right = form_da.dash_margin_right.data
        cfg_graph.margin_top = form_da.dash_margin_top.data
        cfg_graph.margin_bottom = form_da.dash_margin_bottom.data
        flash('Grapheinstellungen f√ºr diese Session gespeichert.')

    return render_template('main/einstellungen.html', title='Einstellungen', form=[form_bb, form_up, form_da])
