[docs]    def __init__(self,
                 username: str = None,
                 password: str = None,
                 apikey: str = None) -> None:
        """Constructor.

        Sets the authentication parameters to connect to the Billbee API.
        """
        if username is not None and password is not None:
            self.basic_auth = (username, password)

        if apikey is not None:
            self.headers = {'X-Billbee-Api-Key': apikey}

        if username is None and password is None and apikey is None:
            username, password, apikey = self.get_billbee_data()
            self.basic_auth = (username, password)
            self.headers = {'X-Billbee-Api-Key': apikey}

        self.page_size = 50



[docs]    def get_customers(self) -> 'list':
        """Get customers.

        Asks the Billbee API for a list of customer JSONs and returns it modified.
        The modification is the additional ZIP code to the customer JSON.
        """
        page = 1
        result = []
        while True:
            payload = {'page': page, 'pageSize': self.page_size}
            try:
                customers = requests.get('https://app.billbee.io/api/v1/customers',
                                         auth=self.basic_auth,
                                         headers=self.headers,
                                         params=payload)
                customers.raise_for_status()
            except requests.exceptions.HTTPError as errh:
                logging.error("Http Error: {}".format(errh))
                return []
            except requests.exceptions.ConnectionError as errc:
                logging.error("Connection Error: {}".format(errc))
                return []
            except requests.exceptions.Timeout as errt:
                logging.error("Timeout Error: {}".format(errt))
                return []
            except requests.exceptions.RequestException as err:
                logging.error("Unknown Request Error: {}".format(err))
                return []

            customers_json = customers.json()['Data']
            for customer in customers_json:
                customer_id = customer['Id']
                payload = {'page': '0', 'pageSize': self.page_size}
                try:
                    customer_address = requests.get('https://app.billbee.io/api/v1/customers/'
                                                    + str(customer_id) + '/addresses/',
                                                    auth=self.basic_auth,
                                                    headers=self.headers,
                                                    params=payload)
                    customer_address.raise_for_status()
                except requests.exceptions.HTTPError as errh:
                    logging.error("Http Error: {}".format(errh))
                    return []
                except requests.exceptions.ConnectionError as errc:
                    logging.error("Connection Error: {}".format(errc))
                    return []
                except requests.exceptions.Timeout as errt:
                    logging.error("Timeout Error: {}".format(errt))
                    return []
                except requests.exceptions.RequestException as err:
                    logging.error("Unknown Request Error: {}".format(err))
                    return []
                customer_address_json = customer_address.json()['Data']
                customer_zip = customer_address_json[0]['Zip']
                customer.update({'Zip': customer_zip})
            result = result + customers_json
            if page == customers.json()['Paging']['TotalPages']:
                break
            page += 1
            time.sleep(1)

        logging.debug('Got {} customers from Billbee'.format(len(result)))
        return result

[docs]    def get_billbee_data(self) -> 'str':
        engine = create_engine(cfg.SQLALCHEMY_DATABASE_URI)

        # metadata = MetaData(engine)
        # users = Table('user', metadata, autoload=True)
        # mapper(User, users)

        session = Session(engine)

        result = session.query(User).filter(User.username == 'thorsten').all()
        bb_apikey = result[0].billbee_apikey
        bb_username = result[0].billbee_username
        bb_password = result[0].billbee_password
        session.close()

        return bb_username, bb_password, bb_apikey



[docs]    def get_customer_names(self) -> 'dict':
        """Get customers.

        Asks the Billbee API for a list of customer JSONs. Then extracts the
        ids and names to return them as a dictionary.
        """
        page = 1
        result = {}
        while True:
            payload = {'page': page, 'pageSize': self.page_size}
            try:
                customers = requests.get('https://app.billbee.io/api/v1/customers',
                                         auth=self.basic_auth,
                                         headers=self.headers,
                                         params=payload)
                customers.raise_for_status()
            except requests.exceptions.HTTPError as errh:
                logging.error("Http Error: {}".format(errh))
                return {}
            except requests.exceptions.ConnectionError as errc:
                logging.error("Connection Error: {}".format(errc))
                return {}
            except requests.exceptions.Timeout as errt:
                logging.error("Timeout Error: {}".format(errt))
                return {}
            except requests.exceptions.RequestException as err:
                logging.error("Unknown Request Error: {}".format(err))
                return {}

            customers_json = customers.json()['Data']
            for customer in customers_json:
                customer_id = customer['Id']
                customer_name = customer['Name']
                result[customer_id] = customer_name
            if page == customers.json()['Paging']['TotalPages']:
                break
            page += 1
            time.sleep(1)

        logging.debug('Got {} customers from Billbee'.format(len(result)))
        return result
